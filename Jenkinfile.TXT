//def ReleaseDir = "c:\inetpub\wwwroot"
String projectName = "Sample_Jenkin_Web_1"
pipeline {
			agent any
			stages {
				stage('Source'){
					steps{
						checkout([$class: 'GitSCM', branches: [[name: '*/master']], doGenerateSubmoduleConfigurations: false, extensions: [], submoduleCfg: [], userRemoteConfigs: [[credentialsId: 'sachinpatil@lsssoftware.com', url: 'https://github.com/sachinpatil261/SampleRepo_2']]])
					}
				}
				stage('Build') {
				steps{
						bat """
						cd ${projectName}						
						dotnet publish -c Release /p:Version=${BUILD_NUMBER} /p:DeployOnBuild=true  /p:WebPublishMethod=FileSystem /p:SkipInvalidConfigurations=true  /p:Platform=\"Any CPU\" /p:DeleteExistingFiles=True /p:publishUrl=c:\\inetpub\\wwwroot
						"""					
						  
					
						}
				}				
			}
}
